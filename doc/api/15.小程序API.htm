小程序使用扩展能力demo（以选择文件为例）：

function secondCallback(args) {
    swan.showModal({
        title: '二级回调',
        content: JSON.stringify(args),
        cancelColor: '#999999',
        confirmColor: '#0099cc',
        success: function (res) {
            if (res.confirm) {
                console.log('用户点击了确定');
            } else if (res.cancel) {
                console.log('用户点击了取消');
            }
        }
    });
}
swan.netdisk.selectFile({maxSelectedNum: 2}, secondCallback)
     .then(res => {
         swan.showToast({
             title: '一级回调',
             duration: 2000
         });
     })
